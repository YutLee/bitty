 56 	 * 对比返回在 array1 中但是不在 array2 中的值。
 57 	 * @param {Array} array1 必须，要被对比的数组
 58 	 * @param {Array} array2 必须，和这个数组进行比较
 59 	 * @return {Array} 返回一个数组，该数组包括了所有在 array1 中但是不在 array2 中的值。
 60 	 * @memberOf _global_
 61 	 */
 62 	function arrayDiff(array1, array2) {
 63 		var that = this;
 64 		if(!isArray(array1)) {
 65 			console.log('参数 array1：' + array1 + '不是数组');
 66 			return false;
 67 		}
 68 		if(!isArray(array2)) {
 69 			console.log('参数 array2：' + array2 + '不是数组');
 70 			return false;
 71 		}
 72 		if(isArray(array1) && isArray(array2)) {
 73 			var newArray = [],
 74 				k1 = 0,
 75 				len1 = array1.length,
 76 				len2 = array2.length;
 77 			for(; k1 < len1; k1++) {
 78 				for(var k2 = 0; k2 < len2; k2++) {
 79 					if(array1[k1] === array2[k2]) {
 80 						break;
 81 					}
 82 					if(k2 === len2 - 1 && array1[k1] !== array2[k2]) {
 83 						newArray.push(array1[k1]);
 84 					}
 85 				}
 86 			}
 87 			return newArray;
 88 		}
